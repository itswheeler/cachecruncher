Formulas used
  • num_blocks = cache_size / block_size
  • num_sets = num_blocks / associativity
  • offset_bits = log₂(block_size)
  • index_bits = log₂(num_sets)
  • tag_bits = address_size − (index_bits + offset_bits)
  • num_tags = num_blocks
  • tag_space (bytes) = num_tags × tag_bits / 8
  • valid_bits (bytes) = num_blocks / 8
  • page_offset_bits = log₂(page_size)
  • virtual_pages = 2^(virtual_address_size − page_offset_bits)
  • physical_pages = physical_memory / page_size
  • VPN_bits = virtual_address_size − page_offset_bits
  • page_table_bits = virtual_pages × (phys_page_bits + protection_bits)
  • tlb_index = log₂(tlb_rows)
  • tlb_tag = VPN_bits − tlb_index
